210def get_moon_phase_panel(observatory, midnight_utc, moon_separation): 

211 #angle of the tilt of the Moon will be different as seen from different latitudes.  

212 #https://astronomy.stackexchange.com/questions/24711/how-does-the-moon-look-like-from-different-latitudes-of-the-earth 

213 #Calculate lunar orbital phase in radians. 

214 

215 sun_midnight = get_body("sun", midnight_utc, location=observatory) 

216 moon_midnight = get_body("moon", midnight_utc, location=observatory) 

217 

218 elongation = sun_midnight.separation(moon_midnight) 

219 moon_phase_angle_inc = np.arctan2(sun_midnight.distance*np.sin(elongation), 

220 moon_midnight.distance - sun_midnight.distance*np.cos(elongation)) 

221 

222 fraction_illuminated = (1 + np.cos(moon_phase_angle_inc))/2.0 

223 fraction_illuminated_percentage = "{:.0%}".format(fraction_illuminated) 

224 angle = np.arctan(np.cos(sun_midnight.dec) * np.sin(sun_midnight.ra - moon_midnight.ra), np.sin(sun_midnight.dec) * np.cos(moon_midnight.dec) - 

225 np.cos(sun_midnight.dec) * np.sin(moon_midnight.dec) * np.cos(sun_midnight.ra - moon_midnight.ra)) #technically, the 2nd parameter calls for arctan2, but we are only interested in the sign of the angle, so it does not matter if we use arctan or arctan2 here. 

226 phase = 0.5 + 0.5 * moon_phase_angle_inc.value * np.sign(angle.value) / np.pi 

227

236 if(phase == 0.25): 236 ↛ 237line 236 didn't jump to line 237 because the condition on line 236 was never true

237 phase_name='first quarter' 

238 phase_image = 'Moon_firstquarter.png' 

239 if(0.25 < phase < 0.5): 239 ↛ 240line 239 didn't jump to line 240 because the condition on line 239 was never true

240 phase_name='waxing gibbous' 

241 phase_image = 'Moon_waxinggibbous.png' 

242 if(phase == 0.5): 242 ↛ 243line 242 didn't jump to line 243 because the condition on line 242 was never true

243 phase_name='full' 

244 phase_image = 'Moon_full.png' 

245 if(0.5 < phase < 0.75): 245 ↛ 246line 245 didn't jump to line 246 because the condition on line 245 was never true

246 phase_name='waning gibbous' 

247 phase_image = 'Moon_waninggibbous.png' 

248 if(phase == 0.75): 248 ↛ 249line 248 didn't jump to line 249 because the condition on line 248 was never true

249 phase_name='last quarter' 

250 phase_image = 'Moon_lastquarter.png' 

251 if(0.75 < phase < 1): 251 ↛ 252line 251 didn't jump to line 252 because the condition on line 251 was never true

252 phase_name='waning crescent' 

253 phase_image = 'Moon_waningcrescent.png' 

254 if(phase == 1): 254 ↛ 255line 254 didn't jump to line 255 because the condition on line 254 was never true

255 phase_name='new moon' 

256 phase_image = 'Moon_new.png' 

257 

258 #Rotate the moon picture counterclockwise by (lat_observatory). 

259 rotation = observatory.lat.value 

260 

261 #html = '<div>'

265 html += f'Moon Phase at {midnight_utc.strftime("%Y-%m-%d %H:%M:%S")} UTC<br>' #TODO/IDEA: Should we distinguish between UTC and local time here? I think UTC is more standard for astronomy, but we could also add the local time in parentheses 

266 html += f'Phase: {phase_name}<br>' 

267 html += f'Illumination: {fraction_illuminated_percentage}<br>' 

268 html += f'separation from moon <br>to object during night: {np.min(moon_separation.degree):.3f} to {np.max(moon_separation.degree):.3f} degree' 

269 #html += '</div>'

271 #this is how it should look like: https://astronomy.stackexchange.com/questions/24711/how-does-the-moon-look-like-from-different-latitudes-of-the-earth 

272 return html
